  def allFilesFound?()
    d = File.expand_path(@source)
    Find.find(d) do |item|
      searcheditem = Utils.src2dest(@cfg, item)
      if !File.exist?(searcheditem)
        puts "allFilesFound?: did not find #{item} in destdir"
        return false 
      end
    end
    
    true
    
  end
