  def dirname(path)
  
    # reuse baseclass' dirname-func 

    result = super(path)
    
    # ...but remove any trailing dots, match pattern /., \. and :.

    # these dots occurs when giving win-dirs like H: that else becomes H:.

    result.gsub!(/.$/,"") if result.match(/[\/\\\:]\.$/)
    
    result

  end
