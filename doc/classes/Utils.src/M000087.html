  def Utils.createConfig(worker, params)
    # use brute force for converting paths to windows
    if RUBY_PLATFORM =~ /mswin/
      params.merge!(params.each_value { |v| v.gsub!("/","\\") })
    end
  
    { "jobs" => 
      [ {
        "worker" => worker,
        "parameters" => params
      } ] 
    }
  end
