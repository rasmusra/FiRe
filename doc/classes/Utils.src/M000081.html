  def Utils.runFiRe(cfg)
    
    # just put the tmp configfile somewhere
    configfile = "test/data/tmp.yml"
    
    # create the configfile
    f = File.open(configfile, "w")
    f.print(cfg.to_yaml)
    f.close

    # add args if not done before
    if !$*.include?("-config")
      $*.push("-config") 
      $*.push(configfile)
    end
    
    # run program
    require 'lib/fire.rb'
    main
    
  end
