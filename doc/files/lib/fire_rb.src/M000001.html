def main
  begin  
  
    ResourceLocator.setLog(Logger["log"])
    FiRe::log.info "Starting up!"
  
    configdata = YAML.load(File.open(ARGV.flags.config.to_s)) 
    
    configdata["jobs"].each { |w| 
      WorkerFactory.create(w).execute
    }
  
  rescue StandardError => e
    log.fatal e
    raise
  end
end
